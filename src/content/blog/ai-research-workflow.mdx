```python
class ResearchFilter:
    def __init__(self):
        self.priority_authors = []
        self.priority_institutions = []
        self.keywords = []

    def score_paper(self, paper):
        score = 0

        # Author reputation
        if paper.author in self.priority_authors:
            score += 10

        # Citation count
        score += min(paper.citations / 100, 5)

        # Keyword relevance
        for keyword in self.keywords:
            if keyword in paper.abstract:
                score += 2

        return score
```
